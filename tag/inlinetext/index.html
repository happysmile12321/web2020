package main

import (
	"github.com/jroimartin/gocui"
	"log"
)

func main() {
	//创建1个只有8种颜色的GUI对象
	g, err := gocui.NewGui(gocui.OutputNormal)
	//如果在构建gocui对象（其实构建的是termbox对象）
	//失败，就返回err.
	if err != nil {
		//日志记录一下
		log.Fatal(err)
	}
	//等待构建这个对象结束，拿到这个对象的指针后
	//就释放构建这个对象所需要申请的资源
	defer g.Close()
	//一些键盘绑定、视图的设置
	//设置视图管理函数，其实就是布局函数的名字。
	g.SetManagerFunc()

	//设置键盘绑定，不过要绑定多个.
	//因为默认只能设置1个绑定，帮顶需要g对象，所以构造
	//1个函数将g穿进去，统一进行注册管理键盘的绑定。
	//注意，如果是“”则代表对全部视图进行绑定。
	//否则，是对给定的视图的名字进行绑定。
	//绑定会使主循环产生事件从而逃离阻塞,进行资源的更新。
	if err := setkeybindings(g); err != nil {
		log.Panic(err)
	}

	//如果主循环返回了错误，而且这个错误还不是
	//正常退出程序这个错误,那么用日志记录一下
	if err := g.MainLoop(); err != nil && err != gocui.ErrQuit {
		log.Fatal(err)
	}
}

//键盘绑定绑定的是视图，或者g对象。
func setkeybindings(g *gocui.Gui) error {

	//总体返回nil，这样,单独的哪个键盘绑定出现了问题就提醒哪个。
	return nil
}

wbr
该HTML wbr元素代表一个字破门良机，在文本中位置的浏览器可以选择断行，但其行规则本来不会创建在该位置休息。 wbr and br.

html4中br表示换行，br表示必须换行。而wbr意思是在浏览器窗口或父级窗口宽度足够的情况下，不换行；在宽度不足的情况下，在加了wbr处主动换行。

这个wbr好像对字符型的语言用处挺大，但是对于想中文这种字，貌似没多大用处。
